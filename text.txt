void use(Objects &object)
	{
		if (!in_vector(items->ReturnObjects(), object))
			std::cout << "The object was not found";
		else
		{
			switch (object)
			{
				case CIGARETTE:
                    regeneration();
					std::cout << "The cigarette has been used\n";
                    break;
				case BEER:
				    std::cout << "The beer has been used";
					std::cout << (shotgun->fire() ? "FULL" : "EMPTY") << std::endl;
					shotgun->del();
                    break;
                case MAGNIFIER:
				    std::cout << "The magnifier has been used\n";
					std::cout << (shotgun->fire() ? "FULL" : "EMPTY") << std::endl;
                    break;
				case KNIFE:
				    std::cout << "The knife has been used\n";
                    shotgun->swap_damage();
                    break;
                default:
                    std::cout << "Unknown object\n";
                    break;
			}
		}
	}
